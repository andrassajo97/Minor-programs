#include <stdio.h>
#include <stdlib.h>
#include <sys/types.h>
#include <sys/ipc.h>
#include <sys/shm.h>
#include <string.h>
#include <signal.h>

#define KEY 1997

typedef struct
{
        char text[100];
        char textek[100];
}Messages;

int main()
{
        printf("CHAT PROGRAM\tKilépéshez küldd el a ""kilépek"" üzenetet!\n");
        printf("\n");

        int shmid;
        char uzenet[100];
        Messages *msg;

        shmid = shmget(KEY, 1024, IPC_CREAT | 00666);
        if (shmid == -1)
                perror("shmget()"), exit(-1);

        msg = (Messages *)shmat(shmid, NULL, IPC_CREAT | 00666);
        if (msg == (void *)-1)
                perror("shmat()"), exit(-1);

        char segit[100];
        strcpy(segit,msg->text);

        while(1)
        {
                printf("O: %s\n",msg->text);
                strcpy(segit,msg->text);

                printf("Te: ");scanf("\t%[^\n]s",uzenet);
                strcpy(msg->textek,uzenet);

                if(strcmp(msg->textek,"kilépek")==0)
                        break;

                while(strcmp(segit,msg->text)==0)
                {
                        wait();
                }
        }

        shmdt(msg);
        shmctl(shmid, IPC_RMID, NULL);
        return 0;
}
